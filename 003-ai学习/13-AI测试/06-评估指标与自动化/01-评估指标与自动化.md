## 三种评估方式的对比

AI 测试的核心挑战是：**输出是自然语言，很难用程序直接判断对错**。

目前有三种主流评估方式：

```text
评估方式          优点                        缺点                      适用场景
─────────────────────────────────────────────────────────────────────────────
人工评估          最准确，能理解细微含义       慢、贵、无法大规模          最终质量基准，抽查
规则评估          快、稳定、成本低             只能检查明确的规则          格式检查、关键词验证
LLM-as-a-Judge   接近人工质量，可自动化        需要设计好评判 Prompt       语义质量、相关性
```

实际项目中，三种方式结合使用：

```text
日常 CI 回归：规则评估（快，0 成本）
                    +
           LLM-as-a-Judge（自动评估质量）
                    ↓
发版前：抽取 10-20% 样本人工复查
```

## 人工评估

### 人工评估标准化

人工评估主观性强，容易产生评审者之间不一致的结果。
解决方法是制定明确的评分标准（Rubric）。

```python
# 人工评估打分卡示例
SCORING_RUBRIC = {
    "accuracy": {
        "description": "回答的事实准确性",
        "levels": {
```

### 人工评估记录表

```python
import csv
from pathlib import Path
from datetime import datetime


def create_manual_eval_sheet(
    cases: list[dict],
    output_file: str = "manual_eval.csv"
):
    """
    生成供人工评估用的 CSV 表格。
    cases 格式：[{"id": "...", "question": "...", "answer": "..."}]
    """
    rows = []
    for case in cases:
        rows.append({
            "id": case["id"],
            "question": case["question"],
            "answer": case["answer"],
            "accuracy_score": "",    # 评审者填写 1-5
            "relevance_score": "",
            "completeness_score": "",
            "comments": "",          # 评审者备注
            "reviewer": "",
            "reviewed_at": ""
        })

    fieldnames = list(rows[0].keys())
    with open(output_file, "w", newline="", encoding="utf-8-sig") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        writer.writerows(rows)

    print(f"人工评估表已生成：{output_file}（共 {len(rows)} 条）")


def load_manual_eval_results(eval_file: str) -> dict:
    """加载人工评估结果，计算各维度平均分"""
    scores = {"accuracy": [], "relevance": [], "completeness": []}

    with open(eval_file, encoding="utf-8-sig") as f:
        reader = csv.DictReader(f)
        for row in reader:
            for dim in scores:
                val = row.get(f"{dim}_score", "").strip()
                if val.isdigit():
                    scores[dim].append(int(val))

    result = {}
    for dim, vals in scores.items():
        if vals:
            result[dim] = {
                "avg": round(sum(vals) / len(vals), 2),
                "count": len(vals)
            }

    return result
```

## 规则评估

规则评估是最稳定、成本最低的评估方式，适合检查明确的格式和内容规则。

### 常用规则评估函数

```python
import json
import re
from typing import Callable


class RuleEvaluator:
    """规则评估器，支持组合多种规则"""

    def __init__(self):
        self.rules: list[tuple[str, Callable]] = []

    def add_rule(self, name: str, fn: Callable[[str], bool]):
        """添加一条规则。fn 接收输出文本，返回是否通过"""
        self.rules.append((name, fn))
        return self   # 支持链式调用

    def evaluate(self, output: str) -> dict:
        """对输出文本运行所有规则"""
        results = {}
        all_pass = True

        for name, fn in self.rules:
            try:
                passed = fn(output)
            except Exception as e:
                passed = False
                print(f"  规则 '{name}' 执行出错：{e}")

            results[name] = passed
            if not passed:
                all_pass = False

        return {"all_pass": all_pass, "rules": results}


# ── 常用规则函数库 ──────────────────────────────────────────

def is_valid_json(output: str) -> bool:
    """检查输出是否为合法 JSON"""
    try:
        json.loads(output)
        return True
    except json.JSONDecodeError:
        return False


def json_has_keys(*required_keys: str):
    """检查 JSON 输出是否包含指定字段"""
    def check(output: str) -> bool:
        try:
            data = json.loads(output)
            return all(k in data for k in required_keys)
        except Exception:
            return False
    return check


def contains_all(*keywords: str):
    """检查输出是否包含所有关键词"""
    def check(output: str) -> bool:
        return all(kw in output for kw in keywords)
    return check


def contains_none(*keywords: str):
    """检查输出中是否不含指定词"""
    def check(output: str) -> bool:
        return not any(kw in output for kw in keywords)
    return check


def length_between(min_len: int, max_len: int):
    """检查输出长度是否在范围内"""
    def check(output: str) -> bool:
        return min_len <= len(output) <= max_len
    return check


def matches_pattern(pattern: str):
    """检查输出是否匹配正则表达式"""
    def check(output: str) -> bool:
        return bool(re.search(pattern, output))
    return check


def starts_with(prefix: str):
    """检查输出是否以指定前缀开头"""
    def check(output: str) -> bool:
        return output.strip().startswith(prefix)
    return check


# ── 使用示例 ──────────────────────────────────────────────

# 测试"输出情感分析 JSON"的接口
sentiment_evaluator = (
    RuleEvaluator()
    .add_rule("是合法JSON",     is_valid_json)
    .add_rule("有label字段",    json_has_keys("label", "score"))
    .add_rule("label值合法",    lambda o: json.loads(o).get("label") in
                                          ("positive", "negative", "neutral"))
    .add_rule("score在0到1之间", lambda o: 0 <= json.loads(o).get("score", -1) <= 1)
)

# 测试"客服回复"的接口
service_evaluator = (
    RuleEvaluator()
    .add_rule("长度合理",    length_between(20, 300))
    .add_rule("有礼貌用语",  contains_all("您好", "感谢"))
    .add_rule("不提竞争对手", contains_none("竞争对手A", "竞争对手B"))
    .add_rule("不含歧视性词语", contains_none("歧视词1", "歧视词2"))
)


def run_rule_tests(evaluator: RuleEvaluator, test_cases: list[dict]):
    passed_count = 0

    for case in test_cases:
        result = evaluator.evaluate(case["output"])
        ok = result["all_pass"]
        if ok:
            passed_count += 1

        status = "✅" if ok else "❌"
        print(f"{status} [{case['id']}]")
        for rule_name, rule_ok in result["rules"].items():
            rule_status = "  ✓" if rule_ok else "  ✗"
            print(f"{rule_status} {rule_name}")

    print(f"\n通过：{passed_count}/{len(test_cases)}")
```

## LLM-as-a-Judge

LLM-as-a-Judge 是用另一个 LLM 来评估目标 LLM 的输出质量。

适合评估语义质量、相关性、流畅度等规则难以量化的维度。

### 设计评判 Prompt 的原则

```text
好的评判 Prompt 应该：

1. 明确评估维度和评分标准
   不要说"评价一下回答好不好"
   要说"从准确性（1-5）和简洁性（1-5）两个维度评分，标准如下..."

2. 提供评分参考示例（Few-shot）
   告诉模型：4分的回答是什么样的，2分的回答是什么样的

3. 要求输出结构化格式
   用 JSON 格式输出分数和理由，方便程序解析

4. 避免位置偏见
   当比较 A 和 B 时，随机调换 A/B 顺序，重复评估取平均

5. 使用更强的模型做裁判
   裁判模型应该比被测模型更强（或同等级）
```

### 单维度评分

```python
import anthropic
import json

client = anthropic.Anthropic()


def llm_judge_single(
    question: str,
    answer: str,
    dimension: str,
    rubric: str,
    judge_model: str = "claude-haiku-4-5"
) -> dict:
    """
    用 LLM 对答案的某一维度打分。

    dimension：评估维度名称，例如 "准确性"
    rubric：评分标准描述
    """
    prompt = f"""
你是一位严格、公正的 AI 回答评估专家。

用户问题：
{question}

AI 回答：
{answer}

请评估上面 AI 回答的【{dimension}】，评分标准如下：
{rubric}

请严格按照以下 JSON 格式输出，不要输出其他任何内容：
{{"score": 1到5的整数, "reason": "简短的评分理由（不超过50字）"}}
"""
    response = client.messages.create(
        model=judge_model,
        max_tokens=128,
        messages=[{"role": "user", "content": prompt}]
    )

    try:
        return json.loads(response.content[0].text)
    except json.JSONDecodeError:
        # 如果解析失败，尝试提取数字
        text = response.content[0].text
        score_match = re.search(r'"score"\s*:\s*(\d)', text)
        score = int(score_match.group(1)) if score_match else 0
        return {"score": score, "reason": "解析失败"}


# 使用示例
result = llm_judge_single(
    question="FastAPI 如何处理请求体？",
    answer="使用 Pydantic 模型定义请求体结构，在路由函数中声明该类型的参数即可自动解析。",
    dimension="准确性",
    rubric="""
5分：完全准确，涵盖核心概念
4分：基本准确，有轻微措辞问题
3分：大体正确，遗漏了重要细节
2分：有明显错误
1分：完全错误
"""
)
print(f"准确性：{result['score']}/5 — {result['reason']}")
```

### 多维度综合评分

```python
EVALUATION_DIMENSIONS = {
    "accuracy": {
        "name": "准确性",
        "rubric": "5=完全正确；4=基本正确有细微偏差；3=部分正确有错误；2=错误较多；1=完全错误"
    },
    "relevance": {
        "name": "相关性",
        "rubric": "5=完全聚焦问题；4=主要相关少量废话；3=部分相关；2=大量离题；1=基本不相关"
    },
    "clarity": {
        "name": "表达清晰度",
        "rubric": "5=表述清晰易懂；4=较清晰；3=有些含糊；2=难以理解；1=表述混乱"
    }
}


def llm_judge_multi(
    question: str,
    answer: str,
    dimensions: dict = EVALUATION_DIMENSIONS,
    judge_model: str = "claude-haiku-4-5"
) -> dict:
    """多维度综合评分，一次调用评估所有维度（节省 Token）"""

    dims_desc = "\n".join(
        f"- {info['name']}：{info['rubric']}"
        for info in dimensions.values()
    )

    keys_desc = ", ".join(f'"{k}": 分数' for k in dimensions)

    prompt = f"""
你是一位严格公正的 AI 回答评估专家。

用户问题：
{question}

AI 回答：
{answer}

请从以下维度对回答打分（每个维度 1-5 分）：
{dims_desc}

严格按照以下 JSON 格式输出，不要输出其他内容：
{{
  {keys_desc},
  "overall_comment": "整体评价（不超过50字）"
}}
"""
    response = client.messages.create(
        model=judge_model,
        max_tokens=256,
        messages=[{"role": "user", "content": prompt}]
    )

    try:
        data = json.loads(response.content[0].text)
        # 计算加权平均分
        scores = [data.get(k, 0) for k in dimensions]
        data["weighted_avg"] = round(sum(scores) / len(scores), 2)
        return data
    except Exception as e:
        return {"error": str(e)}


def run_llm_judge_batch(test_cases: list[dict]) -> list[dict]:
    """批量运行 LLM-as-a-Judge 评估"""
    results = []

    for case in test_cases:
        scores = llm_judge_multi(
            question=case["question"],
            answer=case["answer"]
        )

        avg = scores.get("weighted_avg", 0)
        flag = "✅" if avg >= 3.5 else ("⚠️" if avg >= 2.5 else "❌")
        print(f"{flag} [{case['id']}] 综合得分：{avg}/5")
        print(f"   {scores.get('overall_comment', '')}")

        results.append({"id": case["id"], "scores": scores})

    return results
```

### 避免位置偏见

```python
def judge_with_swap(
    question: str,
    answer_a: str,
    answer_b: str
) -> str:
    """
    比较两个答案，通过交换顺序避免位置偏见。
    返回 "A" / "B" / "tie"
    """

    def compare(q, ans1, ans2, label1, label2) -> str:
        prompt = f"""
请判断以下两个 AI 回答，哪个更好地回答了用户的问题。

用户问题：{q}

回答 {label1}：
{ans1}

回答 {label2}：
{ans2}

只输出一个字母：{label1} 或 {label2}，表示哪个更好。如果一样好，输出 tie。
"""
        resp = client.messages.create(
            model="claude-haiku-4-5",
            max_tokens=10,
            messages=[{"role": "user", "content": prompt}]
        )
        return resp.content[0].text.strip()

    # 正向比较
    result1 = compare(question, answer_a, answer_b, "A", "B")
    # 反向比较（交换顺序）
    result2 = compare(question, answer_b, answer_a, "B", "A")

    # 两次结果一致才认为有效
    if result1 == "A" and result2 == "B":
        return "A"
    elif result1 == "B" and result2 == "A":
        return "B"
    else:
        return "tie"   # 结果不一致，视为平局
```

## 自动化评测集

自动化评测集是一组固定的测试用例，用于定期评估系统质量变化。

### 评测集设计原则

```text
1. 覆盖性
   覆盖所有主要使用场景（用户旅程的每个关键节点）
   覆盖高频问题（Top 20 用户问题）
   覆盖边界情况（空输入、超长输入、混合语言）

2. 代表性
   从真实用户数据中抽取，不要全靠人工构造
   定期更新（每月或每季度）

3. 稳定性
   测试用例本身不随意改动，改动要有记录
   可以扩充，但不要修改已有用例

4. 可追溯性
   每个用例有 ID、创建时间、来源说明
```

### 评测集管理

```python
from dataclasses import dataclass, asdict
from datetime import datetime
import json


@dataclass
class EvalCase:
    id: str
    question: str
    reference_answer: str       # 参考答案（人工写的标准答案）
    category: str               # 分类（如 "产品咨询", "技术问题"）
    source: str                 # 来源（如 "用户日志", "人工构造"）
    created_at: str = ""
    tags: list = None

    def __post_init__(self):
        if not self.created_at:
            self.created_at = datetime.now().isoformat()
        if self.tags is None:
            self.tags = []


class EvalDataset:
    """评测集管理器"""

    def __init__(self, path: str):
        self.path = path
        self.cases: list[EvalCase] = []
        self._load()

    def _load(self):
        try:
            data = json.loads(open(self.path, encoding="utf-8").read())
            self.cases = [EvalCase(**item) for item in data]
        except FileNotFoundError:
            self.cases = []

    def save(self):
        data = [asdict(c) for c in self.cases]
        open(self.path, "w", encoding="utf-8").write(
            json.dumps(data, ensure_ascii=False, indent=2)
        )

    def add(self, case: EvalCase):
        # 检查 ID 是否重复
        if any(c.id == case.id for c in self.cases):
            raise ValueError(f"ID 重复：{case.id}")
        self.cases.append(case)
        self.save()

    def get_by_category(self, category: str) -> list[EvalCase]:
        return [c for c in self.cases if c.category == category]

    def sample(self, n: int, seed: int = 42) -> list[EvalCase]:
        """随机抽样，用于快速回归"""
        import random
        random.seed(seed)
        return random.sample(self.cases, min(n, len(self.cases)))

    def stats(self):
        from collections import Counter
        cats = Counter(c.category for c in self.cases)
        print(f"总用例数：{len(self.cases)}")
        for cat, count in cats.most_common():
            print(f"  {cat}：{count} 条")


# 使用示例
dataset = EvalDataset("eval_dataset.json")

dataset.add(EvalCase(
    id="eval_001",
    question="你们的退款政策是什么？",
    reference_answer="支持7天无理由退款，退款到账3-5个工作日。",
    category="售后政策",
    source="用户日志_2024Q1",
    tags=["退款", "高频"]
))
```

## CI 回归测试

把 AI 测试接入 CI/CD 流水线，每次修改 Prompt 或代码后自动运行评测。

### GitHub Actions 配置

```yaml
# .github/workflows/ai-regression.yml
name: AI 回归测试

on:
  push:
    paths:
      - "prompts/**"         # Prompt 文件变更时触发
      - "src/ai/**"          # AI 相关代码变更时触发
  pull_request:
    paths:
      - "prompts/**"
      - "src/ai/**"

jobs:
  regression:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: 设置 Python 环境
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: 安装依赖
        run: pip install anthropic pytest

      - name: 运行 AI 回归测试
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: python -m pytest tests/ai_regression/ -v --tb=short

      - name: 上传测试报告
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: ai-test-report
          path: reports/
```

```python
PASS_THRESHOLD = 0.80     # 通过率低于 80% 则 CI 失败


def load_test_cases(file: str) -> list[dict]:

def evaluate_output(output: str, expected: dict) -> bool:
    """规则评估（快速，用于 CI）"""
    for kw in expected.get("must_contain", []):

        except Exception:
            return False
    return True


class TestChatAssistantPrompt:

            if evaluate_output(output, case["expected"]):
```

```python
def generate_markdown_report(results: list[dict]) -> str:
    """生成 Markdown 格式报告，可直接贴到 GitHub PR 评论"""
    total = len(results)
    passed = sum(1 for r in results if r.get("passed", False))
    pass_rate = passed / total * 100 if total else 0

    status_emoji = "✅" if pass_rate >= 80 else ("⚠️" if pass_rate >= 60 else "❌")

    lines = [
        f"## {status_emoji} AI 回归测试结果",
        f"",
        f"| 指标 | 结果 |",
        f"|------|------|",
        f"| 通过率 | **{pass_rate:.1f}%**（{passed}/{total}）|",
        f"| 运行时间 | {datetime.now().strftime('%Y-%m-%d %H:%M')} |",
        f"",
    ]

    # 列出失败的用例
    failures = [r for r in results if not r.get("passed", True)]
    if failures:
        lines.append("### ❌ 失败用例")
        lines.append("")
        for r in failures:
            lines.append(f"- **[{r['id']}]** {r.get('question', '')[:50]}")
            for f in r.get("failures", []):
                lines.append(f"  - {f}")
    else:
        lines.append("### ✅ 所有用例通过")

    return "\n".join(lines)
```

## 本节重点

- AI 测试有三种评估方式：人工评估（最准）、规则评估（最快）、LLM-as-a-Judge（平衡）。
- 日常 CI 用规则评估 + LLM-as-a-Judge，发版前抽样人工复查。
- 规则评估用来检查格式、关键词、长度等明确条件，稳定可靠。
- LLM-as-a-Judge 评判时要给出明确的评分标准，并通过交换顺序避免位置偏见。
- 自动化评测集要有覆盖性、代表性和稳定性，不能随意改动已有用例。
- CI 回归测试设置通过率阈值，低于阈值就阻断合并，防止质量下滑。
- 测试报告要可读，方便排查问题，HTML 或 Markdown 均可。
